    \subsection{Docker}
    Turn off all containers
    \begin{minted}{bash}
        docker container stop $(docker container ls -aq)
    \end{minted}

    using a filter
    \begin{minted}{bash}
        docker container prune --filter "until=12h"
    \end{minted}
